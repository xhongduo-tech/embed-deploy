使用 OpenAI NPM 包与 Fetch API
import OpenAI from "openai";

// 1. 向量化调用 (完美兼容 OpenAI SDK)
const openai = new OpenAI({
  baseURL: "http://api-embed.cs.icbc/v1",
  apiKey: "sk-internal" // 必填，任意值均可
});

async function generateEmbedding() {
  const response = await openai.embeddings.create({
    model: "bge-m3",
    input: "测试 Node.js 接入环境",
  });
  console.log(response.data[0].embedding);
}

// 2. 重排调用 (使用原生 Fetch 发起 POST 请求)
async function rerankDocs() {
  const res = await fetch("http://api-embed.cs.icbc/v1/rerank", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "bge-reranker-v2-m3",
      query: "API网关配置",
      documents: ["统一域名访问模式非常优雅。", "今天吃什么？"]
    })
  });
  console.log(await res.json());
}

    check_embedding_ctx_length=False
)

# 1. 对搜索词进行单条向量化
